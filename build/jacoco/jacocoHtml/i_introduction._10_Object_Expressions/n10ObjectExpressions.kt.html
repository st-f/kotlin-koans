fun task10(): List<Int> {
    val arrayList = arrayListOf(1, 5, 2)
    Collections.sort(arrayList, Comp)
    return arrayList
}

object Comp : Comparator<Int> {
    override fun compare(p0: Int, p1: Int): Int {
        return p1 - p0
    }
}
